Listeners.json lists all listeners for every course and section in this format:
{
    "COURSE101": {
        "listeners": [
            {
                is_user
                ID
                section_added
                section_removed
                any_section_opens
                any_honors_opens
                any_online_opens
            }
        ]
        "sections": {
            "SECTION001": [
                is_user
                ID
                open
                close
                prof_update
                timing_room_update
            ]
        }
    }
}
